#include <stdio.h>
#define MAX(a,b) ((a>b)?(a):(b))
int find(int* par, int n)
{
  int p = par[n];

  while (p != par[p])
  {
      par[p] = par[par[p]];
      p = par[p];
  }
  return p;
}

void union_(int* rank, int* par, int n1, int n2)
{
    int p1 = find(par, n1);
    int p2 = find(par, n2);

    if (p1 == p2)
     return;

    if (rank[p1] > rank[p2])
    {
        par[p2] = p1;
        rank[p1] += rank[p2];
    }
    else
    {
        par[p1] = p2;
        rank[p2] += rank[p1]; 
    }
    return;
}

void dfs(int** graph, int m, int r, int c)
{
  fprintf(stdout,"r=%d c=%d\n",r,c);
  if ((r < 0) || (r > m-1) || (c < 0) || (c > m-1))
    return;
  if (graph[r][c] == 1)
  {
     //graph[r][c] = 0;
     dfs(graph,m,r+1,c);
     dfs(graph,m,r-1,c);
     dfs(graph,m,r,c+1);
     dfs(graph,m,r,c-1);
  }
}

int StoneIndex(int**stones, int len, int r, int c)
{
   for (int i=0; i<len; i++)
   {
       if ((stones[i][0] == r) && (stones[i][1] == c))
        return i;
   }
   return -1;
}
