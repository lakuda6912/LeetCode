class Solution:
    def equalPairs(self, grid: List[List[int]]) -> int:
        n = len(grid)
        c1, c2 = [], []
        for i in range(n):
            for j in range(n):
                c1.append(grid[j][i])
            c2.append(c1)
            c1 = []

        res = 0
        for i in range(n):
            for j in range(n):
                if grid[i] == c2[j]:
                    res += 1
        return res            



        return 0
